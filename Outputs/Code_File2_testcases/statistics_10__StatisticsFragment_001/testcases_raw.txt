package com.example.ecodriving.ui.statistics;

import static com.example.ecodriving.MainActivity.ip;

import android.content.Context;
import android.content.SharedPreferences;
import android.graphics.Color;
import android.os.Bundle;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.Toast;

import androidx.annotation.NonNull;
import androidx.fragment.app.Fragment;

import com.example.ecodriving.databinding.FragmentDashboardBinding;
import com.github.mikephil.charting.charts.LineChart;
import com.github.mikephil.charting.components.XAxis;
import com.github.mikephil.charting.components.YAxis;
import com.github.mikephil.charting.data.Entry;
import com.github.mikephil.charting.data.LineData;
import com.github.mikephil.charting.data.LineDataSet;
import com.github.mikephil.charting.formatter.ValueFormatter;

import org.json.JSONArray;
import org.json.JSONException;
import org.json.JSONObject;

import java.io.IOException;
import java.text.SimpleDateFormat;
import java.util.ArrayList;
import java.util.List;
import java.util.Locale;

import okhttp3.MediaType;
import okhttp3.OkHttpClient;
import okhttp3.Request;
import okhttp3.RequestBody;
import okhttp3.Response;

public class StatisticsFragment extends Fragment {

    private FragmentDashboardBinding binding;
    private OkHttpClient client;

    @Override
    public void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        client = new OkHttpClient.Builder().build(); // Initialize OkHttpClient
    }

    @Override
    public View onCreateView(@NonNull LayoutInflater inflater, ViewGroup container, Bundle savedInstanceState) {
        binding = FragmentDashboardBinding.inflate(inflater, container, false);

        // Configura il grafico settimanale
        setupChart(binding.graphWeekly);

        // Carica i dati dal server
        loadDashboardData();

        return binding.getRoot();
    }

    private void loadDashboardData() {
        SharedPreferences sharedPreferences = requireActivity().getSharedPreferences("EcoDrivingPrefs", Context.MODE_PRIVATE);
        String userId = sharedPreferences.getString("user_id", null);

        if (userId == null) {
            Toast.makeText(getActivity(), "Errore: utente non loggato", Toast.LENGTH_SHORT).show();
            return;
        }

        JSONObject postData = new JSONObject();
        try {
            postData.put("user_id", userId);
        } catch (JSONException e) {
            e.printStackTrace();
            return;
        }

        new Thread(() -> {
            MediaType JSON = MediaType.parse("application/json; charset=utf-8");
            RequestBody body = RequestBody.create(postData.toString(), JSON);
            Request request = new Request.Builder()
                    .url("http://" + ip + ":8081/api/getDashboardData")
                    .post(body)
                    .build();

            try (Response response = client.newCall(request).execute()) {
                if (response.isSuccessful()) {
                    String responseData = response.body().string();

                    JSONObject jsonResponse = new JSONObject(responseData);

                    // Estrarre le statistiche principali
                    double totalEmissions = jsonResponse.getDouble("total_emissions");
                    double totalDistance = jsonResponse.getDouble("total_distance");
                    int weeklyTrips = jsonResponse.getInt("weekly_trips");

                    // Aggiornare le statistiche nella UI
                    requireActivity().runOnUiThread(() -> updateStatistics(totalEmissions, totalDistance, weeklyTrips));

                    // Estrarre i dati per il grafico settimanale
                    JSONArray dailyEmissions = jsonResponse.getJSONArray("daily_emissions");

                    List<Entry> weeklyEmissions = new ArrayList<>();
                    List<String> days = new ArrayList<>();
                    for (int i = 0; i < dailyEmissions.length(); i++) {
                        JSONObject dayData = dailyEmissions.getJSONObject(i);

                        if (dayData.has("giorno") && dayData.has("totale_emissioni")) {
                            String day = dayData.getString("giorno");
                            double emission = dayData.getDouble("totale_emissioni");

                            days.add(day);
                            weeklyEmissions.add(new Entry(i, (float) emission));
                        }
                    }

                    // Aggiornare il grafico nella UI
                    requireActivity().runOnUiThread(() -> {
                        updateGraph(binding.graphWeekly, weeklyEmissions, days);
                    });
                } else {
                    showErrorToast("Errore durante il caricamento dei dati");
                }
            } catch (IOException | JSONException e) {
                e.printStackTrace();
                showErrorToast("Errore di connessione");
            }
        }).start();
    }

    private void updateGraph(LineChart chart, List<Entry> emissionsData, List<String> days) {
        LineDataSet dataSet = new LineDataSet(emissionsData, "Emissioni");
        dataSet.setColor(Color.BLUE);
        dataSet.setValueTextColor(Color.BLACK);
        dataSet.setLineWidth(2f);
        dataSet.setCircleColor(Color.RED);
        dataSet.setCircleRadius(4f);
        dataSet.setDrawValues(true);

        LineData lineData = new LineData(dataSet);
        chart.setData(lineData);

        // Configura l'asse X
        XAxis xAxis = chart.getXAxis();
        xAxis.setValueFormatter(new ValueFormatter() {
            @Override
            public String getFormattedValue(float value) {
                int index = Math.round(value);
                if (index >= 0 && index < days.size()) {
                    // Converti la data nel formato giorno/mese/anno
                    try {
                        String inputDate = days.get(index); // Supponendo che sia nel formato "yyyy-MM-dd"
                        SimpleDateFormat inputFormat = new SimpleDateFormat("yyyy-MM-dd", Locale.getDefault());
                        SimpleDateFormat outputFormat = new SimpleDateFormat("dd/MM/yyyy", Locale.getDefault());
                        return outputFormat.format(inputFormat.parse(inputDate));
                    } catch (Exception e) {
                        e.printStackTrace();
                    }
                }
                return "";
            }
        });
        xAxis.setGranularity(1f);
        xAxis.setGranularityEnabled(true);

        // Configura altre proprietà del grafico
        chart.invalidate(); // Forza il rendering
    }


    private void showErrorToast(String message) {
        requireActivity().runOnUiThread(() -> Toast.makeText(getActivity(), message, Toast.LENGTH_SHORT).show());
    }

    private void setupChart(LineChart chart) {
        chart.getDescription().setEnabled(false);
        chart.setTouchEnabled(true);
        chart.setDragEnabled(true);
        chart.setScaleEnabled(true);
        chart.setPinchZoom(true);

        XAxis xAxis = chart.getXAxis();
        xAxis.setPosition(XAxis.XAxisPosition.BOTTOM);
        xAxis.setDrawGridLines(false);

        YAxis leftAxis = chart.getAxisLeft();
        leftAxis.setDrawGridLines(true);

        YAxis rightAxis = chart.getAxisRight();
        rightAxis.setEnabled(false);
    }

    private void updateStatistics(double totalEmissions, double totalDistance, int weeklyTrips) {
        binding.textTotalEmissionsValue.setText(String.format("%.2f kg", totalEmissions));
        binding.textTravelDistanceValue.setText(String.format("%.2f km", totalDistance));
        binding.textAverageTripsValue.setText(String.valueOf(weeklyTrips));
    }

    @Override
    public void onDestroyView() {
        super.onDestroyView();
        binding = null;
    }
}
