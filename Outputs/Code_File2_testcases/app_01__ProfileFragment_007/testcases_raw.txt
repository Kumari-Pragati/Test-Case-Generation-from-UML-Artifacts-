package com.example.ecodriving.ui.Profile;

import static com.example.ecodriving.MainActivity.ip;

import android.content.Context;
import android.content.Intent;
import android.content.SharedPreferences;
import android.os.Bundle;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.Toast;

import androidx.annotation.NonNull;
import androidx.fragment.app.Fragment;

import com.example.ecodriving.LoginActivity;
import com.example.ecodriving.databinding.FragmentProfileBinding;

import org.json.JSONException;
import org.json.JSONObject;

import java.io.IOException;

import okhttp3.MediaType;
import okhttp3.OkHttpClient;
import okhttp3.Request;
import okhttp3.RequestBody;
import okhttp3.Response;

public class ProfileFragment extends Fragment {

    private FragmentProfileBinding binding;
    private OkHttpClient client = new OkHttpClient();

    @Override
    public View onCreateView(@NonNull LayoutInflater inflater,
                             ViewGroup container, Bundle savedInstanceState) {
        binding = FragmentProfileBinding.inflate(inflater, container, false);
        View root = binding.getRoot();

        // Carica i dati utente quando la pagina viene creata
        loadUserData();

        // Bottone per aggiornare i dati del profilo
        binding.btnUpdateProfile.setOnClickListener(v -> updateUserData());

        // Bottone per disconnettersi
        binding.btnLogout.setOnClickListener(v -> logoutUser());

        return root;
    }

    private void loadUserData() {
        // Recupera l'user_id dalle SharedPreferences
        SharedPreferences sharedPreferences = requireActivity().getSharedPreferences("EcoDrivingPrefs", Context.MODE_PRIVATE);
        String userId = sharedPreferences.getString("user_id", null);

        if (userId == null) {
            Toast.makeText(getActivity(), "Errore: utente non loggato", Toast.LENGTH_SHORT).show();
            return;
        }

        // Invia una richiesta GET per recuperare i dati dell'utente
        new Thread(() -> {
            Request request = new Request.Builder()
                    .url("http://"+ip+":8081/api/getUserData?user_id=" + userId)
                    .get()
                    .build();

            try (Response response = client.newCall(request).execute()) {
                if (response.isSuccessful()) {
                    String responseData = response.body().string();
                    JSONObject userData = new JSONObject(responseData);

                    // Popola i campi di input nel thread principale
                    getActivity().runOnUiThread(() -> {
                        try {
                            binding.etProfileName.setText(userData.getString("full_name"));
                            binding.etProfileEmail.setText(userData.getString("email"));
                            binding.etProfilePassword.setText(userData.getString("password"));
                            binding.etCarBrand.setText(userData.getString("car_brand"));
                            binding.etCO2Emission.setText(userData.getString("co2_emission"));
                            binding.etFuelType.setText(userData.getString("fuel_type"));
                        } catch (JSONException e) {
                            e.printStackTrace();
                        }
                    });
                } else {
                    getActivity().runOnUiThread(() -> {
                        Toast.makeText(getActivity(), "Errore durante il recupero dei dati", Toast.LENGTH_SHORT).show();
                    });
                }
            } catch (IOException | JSONException e) {
                e.printStackTrace();
                getActivity().runOnUiThread(() -> {
                    Toast.makeText(getActivity(), "Errore di connessione", Toast.LENGTH_SHORT).show();
                });
            }
        }).start();
    }

    private void updateUserData() {
        String fullName = binding.etProfileName.getText().toString().trim();
        String email = binding.etProfileEmail.getText().toString().trim();
        String password = binding.etProfilePassword.getText().toString().trim();
        String carBrand = binding.etCarBrand.getText().toString().trim();
        String co2Emission = binding.etCO2Emission.getText().toString().trim();
        String fuelType = binding.etFuelType.getText().toString().trim();

        // Controllo validità email
        if (!android.util.Patterns.EMAIL_ADDRESS.matcher(email).matches()) {
            Toast.makeText(getActivity(), "Inserisci un'email valida", Toast.LENGTH_SHORT).show();
            return;
        }

        SharedPreferences sharedPreferences = requireActivity().getSharedPreferences("EcoDrivingPrefs", Context.MODE_PRIVATE);
        String userId = sharedPreferences.getString("user_id", null);

        if (userId == null) {
            Toast.makeText(getActivity(), "Errore: utente non loggato", Toast.LENGTH_SHORT).show();
            return;
        }

        JSONObject postData = new JSONObject();
        try {
            postData.put("user_id", userId);
            postData.put("full_name", fullName);
            postData.put("newEmail", email);
            postData.put("newPassword", password);
            postData.put("car_brand", carBrand);
            postData.put("co2_emission", co2Emission);
            postData.put("fuel_type", fuelType);
        } catch (JSONException e) {
            e.printStackTrace();
        }

        new Thread(() -> {
            MediaType JSON = MediaType.parse("application/json; charset=utf-8");
            RequestBody body = RequestBody.create(postData.toString(), JSON);
            Request request = new Request.Builder()
                    .url("http://" + ip + ":8081/api/updateUser")
                    .post(body)
                    .build();

            try (Response response = client.newCall(request).execute()) {
                if (response.isSuccessful()) {
                    getActivity().runOnUiThread(() -> {
                        Toast.makeText(getActivity(), "Profilo aggiornato con successo", Toast.LENGTH_SHORT).show();
                    });
                } else {
                    getActivity().runOnUiThread(() -> {
                        Toast.makeText(getActivity(), "Errore durante l'aggiornamento", Toast.LENGTH_SHORT).show();
                    });
                }
            } catch (IOException e) {
                e.printStackTrace();
                getActivity().runOnUiThread(() -> {
                    Toast.makeText(getActivity(), "Errore di connessione", Toast.LENGTH_SHORT).show();
                });
            }
        }).start();
    }


    private void logoutUser() {
        // Rimuovi l'user_id dalle SharedPreferences
        SharedPreferences sharedPreferences = requireActivity().getSharedPreferences("EcoDrivingPrefs", Context.MODE_PRIVATE);
        SharedPreferences.Editor editor = sharedPreferences.edit();
        editor.remove("user_id");
        editor.apply();

        Intent intent = new Intent(getActivity(), LoginActivity.class);
        startActivity(intent);
        requireActivity().finish();
    }

    @Override
    public void onDestroyView() {
        super.onDestroyView();
        binding = null;
    }
}
