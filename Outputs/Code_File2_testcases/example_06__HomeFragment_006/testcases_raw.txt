    private void handleButtonClick() {
        if (isTripActive) {
            endTrip(); // Termina il tragitto
        } else {
            startTrip(); // Avvia il tragitto
        }
    }

    private void startTrip() {
        isTripActive = true;
        speedometer.resetTotalDistance();
        if(speedometer.getLoaded()==false){
            speedTextView.setText("Loading...");
        }

        // Cambia l'icona del pulsante per indicare che il tragitto è attivo
        binding.buttonStartTrip.setImageResource(R.drawable.ic_stop_trip); // Modifica l'icona a "stop"
        Toast.makeText(getActivity(), "Tragitto avviato!", Toast.LENGTH_SHORT).show(); // Messaggio di avvio
    }

    private void endTrip() {
        speedometer.setLoaded(false);
        speedTextView.setText("Avvia il tragitto");
        isTripActive = false;

        // Cambia l'icona del pulsante per indicare che il tragitto è terminato
        binding.buttonStartTrip.setImageResource(R.drawable.ic_start_trip);

        saveTripToDatabase(); // This will handle the refresh after successful save
        Toast.makeText(getActivity(), "Tragitto terminato!", Toast.LENGTH_SHORT).show();
    }

    private void loadDailyCO2() {
        SharedPreferences sharedPreferences = requireActivity().getSharedPreferences("EcoDrivingPrefs", Context.MODE_PRIVATE);
        String userId = sharedPreferences.getString("user_id", null);

        if (userId == null) {
            Toast.makeText(getActivity(), "Errore: utente non loggato", Toast.LENGTH_SHORT).show();
            return;
        }

        String currentDate = new SimpleDateFormat("yyyy-MM-dd", Locale.getDefault()).format(new Date());

        JSONObject postData = new JSONObject();
        try {
            postData.put("user_id", userId);
            postData.put("date", currentDate);
        } catch (JSONException e) {
            e.printStackTrace();
        }

        new Thread(() -> {
            MediaType JSON = MediaType.parse("application/json; charset=utf-8");
            RequestBody body = RequestBody.create(postData.toString(), JSON);
            Request request = new Request.Builder()
                    .url("http://"+ip+":8081/api/getDailyCO2")
                    .post(body)
                    .build();

            try (Response response = client.newCall(request).execute()) {
                if (response.isSuccessful()) {
                    String responseData = response.body().string();
                    JSONObject jsonResponse = new JSONObject(responseData);
                    double totalCO2 = jsonResponse.getDouble("total_co2");
                    getActivity().runOnUiThread(() -> updateCO2Display(totalCO2));
                } else {
                    getActivity().runOnUiThread(() ->
                            Toast.makeText(getActivity(), "Errore durante il caricamento delle emissioni", Toast.LENGTH_SHORT).show());
                }
            } catch (IOException | JSONException e) {
                e.printStackTrace();
                getActivity().runOnUiThread(() ->
                        Toast.makeText(getActivity(), "Errore di connessione", Toast.LENGTH_SHORT).show());
            }
        }).start();
    }

    private void saveTripToDatabase() {
        double distance = speedometer.getTotalDistance();
        speedometer.resetTotalDistance();
        double co2 = (distance*co2gKM/1000);
        String userId = getUserIdFromPreferences();
        if (userId == null) {
            Toast.makeText(getActivity(), "Errore: Utente non loggato!", Toast.LENGTH_SHORT).show();
            return;
        }
        JSONObject tripData = new JSONObject();
        try {
            tripData.put("user_id", userId);
            tripData.put("distance", distance);
            tripData.put("co2", co2);
        } catch (JSONException e) {
            e.printStackTrace();
            return;
        }
        new Thread(() -> {
            MediaType JSON = MediaType.parse("application/json; charset=utf-8");
            RequestBody body = RequestBody.create(tripData.toString(), JSON);
            Request request = new Request.Builder()
                    .url("http://" + ip + ":8081/api/addTrip")
                    .post(body)
                    .build();

            try (Response response = client.newCall(request).execute()) {
                if (response.isSuccessful()) {
                    requireActivity().runOnUiThread(() -> {
                        Toast.makeText(getActivity(), "Tragitto salvato con successo!", Toast.LENGTH_SHORT).show();
                        // Refresh the data after successful save
                        loadDailyTrips();
                        loadDailyCO2();
                    });
                } else {
                    requireActivity().runOnUiThread(() ->
                            Toast.makeText(getActivity(), "Errore durante il salvataggio!", Toast.LENGTH_SHORT).show());
                }
            } catch (IOException e) {
                e.printStackTrace();
                requireActivity().runOnUiThread(() ->
                        Toast.makeText(getActivity(), "Errore di connessione!", Toast.LENGTH_SHORT).show());
            }
        }).start();
    }
