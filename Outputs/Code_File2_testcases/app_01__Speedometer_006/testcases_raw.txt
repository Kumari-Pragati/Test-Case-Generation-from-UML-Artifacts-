package com.example.ecodriving;

import static com.example.ecodriving.ui.home.HomeFragment.isTripActive;

import android.app.Activity;
import android.content.Context;
import android.os.Handler;

import android.Manifest;
import android.content.pm.PackageManager;
import android.location.Location;
import android.widget.TextView;

import androidx.core.app.ActivityCompat;

import com.google.android.gms.location.FusedLocationProviderClient; // Import for Fused Location
import com.google.android.gms.location.LocationCallback;
import com.google.android.gms.location.LocationRequest;
import com.google.android.gms.location.LocationResult;
import com.google.android.gms.location.LocationServices; // Import for Location Services
import com.google.android.gms.location.Priority;

public class Speedometer {
    private static Speedometer instance;
    private float totalDistanceMetri; // Variabile per tenere traccia della distanza totale

    private TextView locationTextView; // TextView per visualizzare la posizione
    private TextView speedTextView; // TextView per visualizzare la velocità

    private FusedLocationProviderClient fusedLocationClient; // Client per la posizione
    private Handler handler; // Handler per aggiornamenti periodici

    private Location previousLocation; // Per memorizzare la posizione precedente
    private double lastUpdateTime; // Per memorizzare il tempo dell'ultimo aggiornamento

    private Context context; // Contesto dell'applicazione
    private Activity activity; // Riferimento all'attività

    private boolean loaded;

    public double getTotalDistance(){
        return totalDistanceMetri / 1000; // manda al DB in km
    }

    public void start(){
        getCurrentLocation(); // Richiesta di aggiornamenti sulla posizione
    }

    private Speedometer(Context context, Activity activity, TextView speedTextView) {
        loaded = false;
        this.context=context;  // Assegna il contesto dell'applicazione
        this.activity=activity;  // Assegna l'attività corrente
        this.speedTextView=speedTextView;  // Assegna il riferimento a speedTextView
        this.locationTextView=locationTextView;  // Assegna il riferimento a locationTextView

        totalDistanceMetri =0.0f;  // Inizializza la distanza totale a zero
        handler=new Handler();  // Inizializza l'handler per gli aggiornamenti periodici

        fusedLocationClient= LocationServices.getFusedLocationProviderClient(context);  // Inizializza il client della posizione
    }

    public static synchronized Speedometer getInstance(Context context, Activity activity, TextView speedTextView) {
        if(instance == null){
            instance=new Speedometer(context, activity, speedTextView);  // Crea una nuova istanza se non esiste già
        }
        return instance;  // Restituisce l'istanza corrente di Speedometer
    }

    private LocationRequest createLocationRequest() {
        return new LocationRequest.Builder(Priority.PRIORITY_HIGH_ACCURACY, 500)  // Intervallo di aggiornamento di 500 ms
                .setMinUpdateIntervalMillis(250)  // Intervallo minimo di aggiornamento di 250 ms
                .build();  // Costruisce la richiesta di posizione
    }

    private void getCurrentLocation() {
        if(ActivityCompat.checkSelfPermission(context, Manifest.permission.ACCESS_FINE_LOCATION) != PackageManager.PERMISSION_GRANTED &&
                ActivityCompat.checkSelfPermission(context, Manifest.permission.ACCESS_COARSE_LOCATION) != PackageManager.PERMISSION_GRANTED){
            ActivityCompat.requestPermissions(activity, new String[]{Manifest.permission.ACCESS_FINE_LOCATION}, 1);  // Richiede i permessi se non concessi
            return;  // Esci dal metodo se i permessi non sono stati concessi
        }

        LocationRequest locationRequest=createLocationRequest();  // Crea la richiesta di posizione

        fusedLocationClient.requestLocationUpdates(locationRequest, new LocationCallback() {
            @Override
            public void onLocationResult(LocationResult locationResult) {
                if(locationResult == null) return;

                for(Location location : locationResult.getLocations()){
                    if(location != null){
                        calculateSpeed(location);  // Calcola la velocità in base alla nuova posizione
                    }
                }
            }
        }, null);
    }

    private void calculateSpeed(Location currentLocation) {
        loaded = true;
        double BASE_DISTANCE = 5; // Distanza base per calcolare la velocità
        double DISTANCE_THRESHOLD  = BASE_DISTANCE;
        if (previousLocation != null) {
            double distance = previousLocation.distanceTo(currentLocation); // Calcola la distanza percorsa
            long currentTime = System.currentTimeMillis();
            double timeElapsed = currentTime - lastUpdateTime;

            if (timeElapsed > 0 && isTripActive) {
                double speed = (distance / timeElapsed) * 1000; // Velocità in m/s
                double kmh = speed * 3.6; // Converti in km/h

                if (timeElapsed > 1200 && timeElapsed < 10000){
                    //DISTANCE_THRESHOLD = BASE_DISTANCE * ((timeElapsed % 1000)*0.0001+(timeElapsed / 1000));
                    DISTANCE_THRESHOLD = BASE_DISTANCE + (timeElapsed *0.001);
                } else if(timeElapsed <=1200){
                    DISTANCE_THRESHOLD = BASE_DISTANCE+0.5;
                }else{
                    DISTANCE_THRESHOLD = BASE_DISTANCE * (timeElapsed *0.001);
                }

                if (kmh < 500 && distance > DISTANCE_THRESHOLD) {
                    totalDistanceMetri += distance; // Aggiorna la distanza totale
                    speedTextView.setText(String.format("%.3f km/h \nDistanza %.3f km", kmh, totalDistanceMetri / 1000));
                } else {
                    speedTextView.setText(String.format("0.0 km/h \nDistanza %.3f km", totalDistanceMetri / 1000));
                }
            }else{
                speedTextView.setText(String.format("Avvia il tragitto", isTripActive));
            }
        }

        previousLocation = currentLocation; // Aggiorna la posizione precedente
        lastUpdateTime = System.currentTimeMillis(); // Aggiorna il tempo dell'ultimo aggiornamento
    }

    public void resetTotalDistance() {
        totalDistanceMetri = 0;
    }

    public boolean getLoaded() {
        return loaded;
    }

    public void setLoaded(boolean b) {
        loaded = b;
    }
}
